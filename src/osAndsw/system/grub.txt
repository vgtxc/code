grub命令:
	help {command}:帮助信息
	ls {path} -al:列出所有文件及文件夹信息
		eg:
			查看根目录信息->ls /
	set {envArg}={path}:设置环境变量envArg的值为path
		eg:
			set root={path}:设置path为根目录
	chainloader {path}:加载kernel内核(.efi文件)
	/:grub默认的根路径
	boot:启动kernel

grub启动windows:
	01.查找efi引导所在的分区
		分区包含的标记文件夹->efi
	02.设置root路径
		# 设置根路径
		set root={efi引导分区}	或	set root={efi引导分区}
		# 查看是否设置成功
		ls	/
		eg:
			set root=(hd0,gpt1)
	03.寻找.efi引导文件
		chainloader +1 	或	chainloader /EFI/Microsoft/Boot/bootmgfw.efi
	04.启动
		boot

grub启动linux:
	01.查找linux系统所在的分区
    	分区包含的标记文件夹->boot;	bin;	dev
	02.指定Linux内核,指定/所在分区
		linux {linux系统内核} ro text
		set root={linux系统分区}
		eg:
			linux /boot/vmlinuz-4.8.0-36-generic
			set root=/dev/sda2
	03.指定初始化文件
		initrd {初始化文件}
		eg:
			initrd	/boot/initrd.img-4.8.0-36-generic
	04.启动
		boot
